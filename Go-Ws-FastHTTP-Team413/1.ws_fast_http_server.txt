package main

import (
	"websocket-client/router/app"
	"websocket-client/router/handlers"
)

func main() {
	a1 := &app.App{}

	hd := handlers.NewHD()
	a1.Router = a1.CreateRouterAndSetRoutes(hd)
	go hd.WebsocketSendHC()
	a1.Run()
}
